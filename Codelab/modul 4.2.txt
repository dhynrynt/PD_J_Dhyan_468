#include <stdio.h>

int main() {
    char str[100];
    int i, j;
    int unit;

    printf("masukkan sebuah string: ");
    gets(str);

    printf("unit karakter: ");

    for (i = 0; str[i] != '\0'; i++) {
        unit = 1;

        for (j = 0; j < i; j++) {
            if (str[i] == str[j]) {
                unit = 0;
                break;
            }
        }

        if (unit) {
            printf("%c ", str[i]);
        }
        
    }

    printf("\n");

    return 0;
}
